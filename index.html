Scss. Миксины

                @mixin background($color) {
                  background: $color;
                }
                  
                .block {
                  @include background(red);
                }
            
                .block {
                  background: red;
                }

Scss. Циклы

                $sizes: 40px, 50px;
                  
                @each $size in $sizes {
                  .block_size_#{$size} {
                     width: $size;
                     height: $size;
                  }
                }
            
                .block_size_40px {
                  width: 40px;
                  height: 40px;
                }
                  
                .block_size_50px {
                  width: 50px;
                  height: 50px;
                }
